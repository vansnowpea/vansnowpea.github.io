14、可实现定ss5密码提醒的脚本，实际他的作用并不大，这是因为当你使用能正常使用youtube的时候，表示ss5正常着，但如果打不开了，八成是密码更新了， 去对应网站手动更新下即可。但既然写了就记录下来，也许对谁有用。

import requests
from lxml import etree
import smtplib
from email.mime.text import MIMEText
from email.header import Header
import copy
import time

# define the time gap for checking in seconds
# 3600 equals to 1 hour
time_gap = 3600
url = 'http://www.ishadowsocks.org/'
html = requests.get(url).content.decode('utf-8')
selector = etree.HTML(html)
password = selector.xpath('//*[@id="free"]/div/div[2]/div[3]/h4[3]/text()')[0]
password = password.split(':')[-1]
print('password is :', password)
temp_password = copy.deepcopy(password)

def check_update():

    if temp_password == password:
        return password,False
    elif temp_password != password:
        return password,True


def send_email(htm):
    # send email to notice new WestWorld is coming
    sender = 'xxxxxxxx@163.com'
    receiver = 'xxxxxxxx@qq.com,xxxxxxxx@163.com'
    subject = 'SS5密码有更新！'
    smtpserver = 'smtp.163.com'
    username = 'xxxxxxxx@163.com'
    password = 'xxxxxxxx'
    msg = MIMEText(htm, 'html', 'utf-8')
    msg['Subject'] = Header(subject, 'UTF-8')
    msg['From'] = sender
    msg['To'] = ','.join(receiver)
    smtp = smtplib.SMTP()
    smtp.connect(smtpserver)
    smtp.login(username, password)
    smtp.sendmail(sender, receiver, msg.as_string())
    smtp.quit()

if __name__ == '__main__':
    password, update = check_update()
    while 1:
        if update == True:
            send_email(password)
        temp_password = password
        time.sleep(time_gap)
